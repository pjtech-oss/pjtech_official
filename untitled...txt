    <!-- Modal -->
    <div class="modal fade" id="selectMethodModal" tabindex="-1" aria-labelledby="selectMethodModalLabel" aria-hidden="true">
        <div class="modal-dialog">
            <div class="modal-content">
                <div class="modal-header">
                    <h5 class="modal-title" id="selectMethodModalLabel">Select Sending Method</h5>
                    <button type="button" class="btn-close" data-bs-dismiss="modal" aria-label="Close"></button>
                </div>
                <div class="modal-body">
                    <button class="btn btn-primary w-100 mb-2" onclick="sendMessage('whatsapp')">Send via WhatsApp</button>
                    <button class="btn btn-secondary w-100" onclick="sendMessage('email')">Send via Email</button>
                </div>
            </div>
        </div>
    </div>

    <script>
        function openModal() {
            var myModal = new bootstrap.Modal(document.getElementById('selectMethodModal'), {
                keyboard: false
            });
            myModal.show();
        }

        function sendMessage(method) {
            // Get form values
            var name = document.getElementById('name').value;
            var email = document.getElementById('email').value;
            var subject = document.getElementById('subject').value;
            var message = document.getElementById('message').value;

            if (method === 'whatsapp') {
                // Construct the WhatsApp message URL
                var whatsappURL = 'https://wa.me/+2347083023292?text=';
                var messageBody = encodeURIComponent('Name: ' + name + '\nEmail: ' + email + '\nSubject: ' + subject + '\nMessage: ' + message);
                whatsappURL += messageBody;

                // Open WhatsApp in a new tab/window
                window.open(whatsappURL, '_blank');
            } else if (method === 'email') {
                // Construct the mailto URL
                var mailtoURL = 'mailtoolioxxayo@gmail.com?subject=' + encodeURIComponent(subject) + '&body=' + encodeURIComponent('Name: ' + name + '\nEmail: ' + email + '\nMessage: ' + message);

                // Open the mail client
                window.location.href = mailtoURL;
            }

            // Hide the modal
            var myModalEl = document.getElementById('selectMethodModal');
            var modal = bootstrap.Modal.getInstance(myModalEl);
            modal.hide();
        }
    </script>
